#include <stdio.h>
#include <stdlib.h>

// Programa C: Ref_002_soma_de_dois_ints

// Calcula a soma de dois inteiros digitados

/*
	Dado que se trata de um programa que vai manipular dados, precisamos de:

	1. Declarar as variáveis

	2. Ler os dados digitados

	3. Calcular a soma dos dados digitados

	4. Mostrar no ecrã o resultado
*/


int main()
{
	int n1, n2, soma;

	printf("\n Digite o Valor 1: ");
	fflush(stdin);
	scanf("%d", &n1);

	printf("\n Digite o Valor 2: ");
	fflush(stdin);
	scanf("%d", &n2);

	soma = n1 + n2;

	printf("\n\nA soma de %d com %d = %d", n1, n2, soma);
	
	printf("\n\n");	
	system("PAUSE");
	
	return 0;
}

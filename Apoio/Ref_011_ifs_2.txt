#include <stdio.h>
#include <stdlib.h>

// Programa C: Ref_011_ifs_2

/* Outro exemplo da estrutura if

   Nota:
   
   Programa que diz o dia da semana por extenso
   em resposta a um valor digitado de 1 a 7.

   Nesta versão vamos usar ifs independentes - 
   que não é boa prática para este programa, 
   já que todos os casos terão que ser testados 
   conforme perceberão bem em breve.

   O objetivo do programa está em perceber-se que,
   embora o programa funcione, deve haver uma forma
   mais eficiente a uasr neste tipo de programas. 
  
   Portanto, brevemente vamos implementar este programa 
   de uma forma eficiente-otimizada.
*/

int main()
{
	int dia;

	printf("\n\nDigite um dia da semana de 1 a 7: ");
	fflush(stdin);
	scanf("%d", &dia);
	
	if (dia == 1)
	{
		printf("\n\nDomingo");
	}
	
	if (dia == 2)
	{
		printf("\n\nSegunda-feira");
	}
	
	if (dia == 3)
	{
		printf("\n\nTerca-feira");
	}

	if (dia == 4)
	{
		printf("\n\nQuarta-feira");
	}

	if (dia == 5)
	{
		printf("\n\nQuinta-feira");
	}

	if (dia == 6)
	{
		printf("\n\nSexta-feira");
	}

	if (dia == 7)
	{
		printf("\n\nSabado");
	}

	if ((dia < 1) || (dia > 7))
	{
		printf("\n\n dia errado");
	}
	
	printf("\n\n");
	system("PAUSE");
	
	return 0;
}

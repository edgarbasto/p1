#include <stdio.h>
#include <stdlib.h>

// Programa C: Ref_012_ifs_3

/* 
   Exemplo da estrutura if else if

   Programa que diz o dia da semana por extenso
   em resposta a um valor digitado de 1 a 7.

   Nesta versão, vamos usar a estrutura if else if.

   Esta é a maneira mais adequada para este tipo de programa já que
   nesta estrutura o teste termina logo que seja encontrado um caso 
   verdadeiro.
*/

int main()
{
	int dia;

	printf("\n\nDigite um dia da semana de 1 a 7: ");
	fflush(stdin);
	scanf("%d", &dia);
	
	if (dia == 1)
	{
		printf("\n\nDomingo");
	}
	else if (dia == 2)
	{
		printf("\n\nSegunda-feira");
	}
	else if (dia == 3)
	{
		printf("\n\nTerca-feira");
	}
	else if (dia == 4)
	{
		printf("\n\nQuarta-feira");
	}
	else if (dia == 5)
	{
		printf("\n\nQuinta-feira");
	}
	else if (dia == 6)
	{
		printf("\n\nSexta-feira");
	}
	else if (dia == 7)
	{
		printf("\n\nSabado");
	}
	else 
	{
		printf("\n\n dia errado");
	}
	
	printf("\n\n");
	system("PAUSE");
	
	return 0;
}

#include <stdio.h>
#include <stdlib.h>

// Programa C: Ref_010_ifs_1

// Indica se um valor digitado é ou não negativo.

int main()
{
	float val;

	printf("\n\nDigite um valor: ");
	fflush(stdin);
	scanf("%f", &val);
	
	if (val >= 0)
	{
		printf("\n\nValor = %4.2f = Nao Negativo", val);
	}
	else
	{
		printf("\n\nValor = %4.2f = Negativo", val);
	}
	
	printf("\n\n\n");
	system("PAUSE");
	
	return 0;
}

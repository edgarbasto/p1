#include <stdio.h>
#include <stdlib.h>

// Programa C: Ref_006_testa_overflow

// Testa overflow
// capacidade de um short: de -32768 a +32767

int main()
{
	short x;
	
	printf("\n\n\t\tTesta overflow para um short\n\n");
	
	printf("\tDigite um Valor inteiro curto: ");	
	fflush(stdin);		
	scanf("%hd", &x);
	
	printf("\n\n\tO Valor atual = %hd", x);
	
	printf("\n\n\n");
	
	system("pause");
	
	return 0;
}

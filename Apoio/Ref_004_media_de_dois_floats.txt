#include <stdio.h>
#include <stdlib.h>

// Programa C: Ref_004_media_de_dois_floats

// Calcula a média de dois floats digitados

/*
	Dado que se trata de um programa que vai manipular dados, precisamos de:

	1. Declarar as variáveis - do tipo float

	2. Ler os dados digitados

	3. Calcular a média dos dados digitados

	4. Mostrar no ecrã o resultado
*/

int main()
{
	float n1, n2, soma, media;

	printf("\n Digite o Valor 1: ");
	fflush(stdin);
	scanf("%f", &n1);

	printf("\n Digite o Valor 2: ");
	fflush(stdin);
	scanf("%f", &n2);

	soma = n1 + n2;

	media = soma / 2;

	printf("\n\nA Media = %4.2f", media);
	
	printf("\n\n");
    	system("PAUSE");
	
	return 0;
}

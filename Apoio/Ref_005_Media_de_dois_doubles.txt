#include <stdio.h>
#include <stdlib.h>

// Programa C: Ref_005_Media_de_dois_doubles

// Calcula a média de dois doubles digitados

// Observar o especificador do tipo de dados usado neste programa: lf (long float)


int main()
{
	double n1, n2, soma, media;

	printf("\n Digite o Valor 1: ");
	fflush(stdin);
	scanf("%lf", &n1); // lf significa long float

	printf("\n Digite o Valor 2: ");
	fflush(stdin);
	scanf("%lf", &n2);

	soma = n1 + n2;

	media = soma / 2;

	printf("\n\nA Media = %4.2lf", media);
	
	printf("\n\n\n");
    	system("PAUSE");
	
	return 0;
}
